<script>
function submitQuiz() {
    const answers = {
        q1: "Eple",
        q2: "Bok",
        q3: "Hus"
    };
    let score = 0;
    for (const q in answers) {
        const selected = document.querySelector(`input[name="${q}"]:checked`);
        if (selected && selected.value === answers[q]) {
            score++;
        }
    }
    document.getElementById("result").innerText = "You scored " + score + " out of 3.";
}
</script>
